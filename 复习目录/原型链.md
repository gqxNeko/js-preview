# 原型链定义
* 个人理解：多个构造函数相互联系，从上至下形成父子关系，这个原型称之为原型链
```javascript
Grand.prototype.lastName = 'guo'
function Grand(){

}
var grand = new Grand()
Father.prototype = Grand
function Father(){
  this.hobbit = 'drink'
}
var father = new Father()
Son.prototype = Father
function Son(){
  this.age = '23'
}
var son = new Son()
son.hobbit//drink
son.lastName//guo
grand.num = 100//非继承值
father.num//undefined
son.num//undefined
```
